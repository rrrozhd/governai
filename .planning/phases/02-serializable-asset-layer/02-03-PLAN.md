    async def put(self, state: RunState) -> None:
        _validate_state(state)
        existing = self._state.get(state.run_id)
        if existing is not None and existing.epoch >= state.epoch:
            raise StateConcurrencyError(

    5. Refactor `RedisRunStore.put()` to use WATCH/MULTI/EXEC (per D-11, D-12):
    ```python
    async def put(self, state: RunState) -> None:
        _validate_state(state)
        client = await self._client()
        run_key = self._key(state.run_id)
        checkpoint_id = state.checkpoint_id or str(uuid.uuid4())
        retries = 3
        delay = 0.05

        for attempt in range(retries + 1):
            async with client.pipeline(transaction=True) as pipe:
                try:
                    await pipe.watch(run_key)
                    raw = await pipe.get(run_key)
                    if raw is not None:
                        text = self._decode_text(raw)
                        if text:
                            existing = RunState.model_validate_json(text)
                            if existing.epoch >= state.epoch:
                                raise StateConcurrencyError(
                                    f"Stale write for run {state.run_id}: "
                                    f"store epoch={existing.epoch}, write epoch={state.epoch}"
                                )
                            state.epoch = existing.epoch + 1
                    else:
                        state.epoch = 1
                    state.checkpoint_id = checkpoint_id
                    payload = state.model_dump_json()
                    pipe.multi()
                    if self.ttl_seconds is None:
                        pipe.set(run_key, payload)
                    else:
                        pipe.set(run_key, payload, ex=int(self.ttl_seconds))
                    cp_key = self._checkpoint_key(checkpoint_id)
                    if self.ttl_seconds is None:
                        pipe.set(cp_key, payload)
                    else:
                        pipe.set(cp_key, payload, ex=int(self.ttl_seconds))
                    pipe.rpush(
                        self._thread_checkpoint_index_key(state.thread_id),
                        checkpoint_id,
                    )
                    await pipe.execute()
                    break
                except StateConcurrencyError:
                    raise
                except Exception as exc:
                    if "WatchError" in type(exc).__name__ or (
                        hasattr(exc, '__class__') and exc.__class__.__name__ == "WatchError"
                    ):
                        if attempt >= retries:
                            raise StateConcurrencyError(
                                f"Optimistic lock conflict on run {state.run_id} "
                                f"after {retries} retries"
                            ) from None
                        await asyncio.sleep(delay * (2 ** attempt))
                    else:
                        raise
        await self._maybe_expire(self._thread_checkpoint_index_key(state.thread_id))
        await self._record_thread_run(state.thread_id, state.run_id)
    ```

    NOTE: The WatchError catch uses name-based detection because `from redis.exceptions import WatchError` should be imported conditionally. Better approach: add `try: from redis.exceptions import WatchError except ImportError: WatchError = None` at the module top and catch it directly:
    ```python
    try:
        from redis.exceptions import WatchError as _WatchError
    except ImportError:
        _WatchError = type(None)
    ```
    Then in the except clause: `except _WatchError:` (after the StateConcurrencyError catch).

    Actually the cleanest approach: import WatchError inside the method or at module level with a try/except, then catch it directly. Use:
    ```python
    try:
        from redis.exceptions import WatchError
    except ImportError:
        class WatchError(Exception): pass  # type: ignore[no-redef]
    ```
    at the top of the file, then catch `WatchError` directly in the retry loop.
